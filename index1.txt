        // --- 2. TEMPLATE GENERATOR ENGINE ---
        
        const situations = [

        const subjects = ["Ali", "Siti", "Ah Meng", "Raju", "Mei Ling", "David", "Aminah"];

        function generateAIQuestion(unitId, index) {
            const unit = units[unitId];
            const subject = subjects[index % subjects.length];
            const situation = situations[index % situations.length];
            
            // Imej telah dibuang seperti yang diminta
            const imgUrl = null;

            const templates = [
                {
                    q: `${subject} berada di ${situation}. Bagaimanakah dia patut menunjukkan sikap ${unit.title}?`,
                    correct: `Mengamalkan sikap ${unit.value} dengan ikhlas`,
                    wrongs: ["Membuat bising dan tidak peduli", "Mementingkan diri sendiri", "Marah kepada semua orang"]
                },
                {
                    q: `Mengenai nilai ${unit.title}, apakah kebaikan mengamalkan sikap ini?`,
                    correct: "Hidup menjadi lebih aman dan harmoni",
                    wrongs: ["Dibenci oleh rakan-rakan", "Menyusahkan ibu bapa", "Membazirkan masa sahaja"]
                },
                {
                    q: `Jika rakan ${subject} tidak mengamalkan nilai ${unit.title}, apakah yang patut ${subject} lakukan?`,
                    correct: "Menegurnya dengan sopan dan baik",
                    wrongs: ["Memukulnya dengan kuat", "Mentertawakannya", "Membiarkan sahaja"]
                },
                {
                    q: `Pilih perlakuan yang BENAR tentang nilai ${unit.title}.`,
                    correct: `Sentiasa ${unit.value} dalam kehidupan seharian`,
                    wrongs: ["Hanya berbuat baik apabila dipandang orang", "Berpura-pura baik", "Mengharapkan balasan wang"]
                }
            ];

            const t = templates[index % templates.length];
            
            return {
                q: t.q,
                img: imgUrl, // Tiada gambar
                options: shuffleArray([t.correct, ...t.wrongs]),
                ans: t.correct,
                unitTitle: unit.title
            };
        }

        // --- 3. HELPER FUNCTIONS ---

        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        function createQuestionHTML(index, data) {
            const optionLabels = ['A', 'B', 'C', 'D'];
            let optionsHTML = '';
            
            data.options.forEach((opt, i) => {
                optionsHTML += `
                    <div class="flex items-start gap-3 mb-3 break-inside-avoid cursor-pointer hover:bg-blue-50 p-1 rounded transition">
                        <span class="font-bold bg-gray-100 text-gray-700 w-8 h-8 flex items-center justify-center rounded-full border border-gray-300 flex-shrink-0">${optionLabels[i]}</span>
                        <span class="pt-1 text-sm md:text-base">${opt}</span>
                    </div>
                `;
            });

            // Container gambarajah telah dikosongkan
            const imageHTML = '';

            return `
                <div class="question-container page-break bg-white p-4 md:p-5 rounded-2xl shadow-sm border border-indigo-100 break-inside-avoid">
                    <div class="flex justify-between items-center mb-3 border-b border-gray-100 pb-2">
                        <span class="text-xs font-bold text-gray-400 uppercase tracking-wider">Soalan ${index}</span>
                        <span class="bg-indigo-100 text-indigo-700 text-[10px] font-bold px-2 py-1 rounded-full no-print">${data.unitTitle}</span>
                    </div>
                    ${imageHTML}
                    <p class="text-gray-800 mb-4 font-semibold text-base md:text-lg comic-font leading-relaxed">${data.q}</p>
                    <div class="pl-1">
                        ${optionsHTML}
                    </div>
                </div>
            `;
        }

        // --- 4. CORE FUNCTIONS ---

        let currentQuestionsHTML = "";
        let currentUnitTitle = "";

        function generatePaper() {
            const selector = document.getElementById('unitSelector');
            const selectedUnit = parseInt(selector.value);
            const unitTitle = units[selectedUnit].title;
            currentUnitTitle = `TAJUK: UNIT ${selectedUnit} - ${unitTitle.toUpperCase()}`;
            
            document.getElementById('loading').classList.remove('hidden');
            document.getElementById('questions-list').innerHTML = '';
            
            setTimeout(() => {
                let htmlContent = '';
                for (let i = 1; i <= 25; i++) {
                    const questionData = generateAIQuestion(selectedUnit, i - 1);
                    htmlContent += createQuestionHTML(i, questionData);
                }

                currentQuestionsHTML = htmlContent;
                document.getElementById('questions-list').innerHTML = htmlContent;
                document.getElementById('loading').classList.add('hidden');
            }, 100);
        }
